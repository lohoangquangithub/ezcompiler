async function runCode() { const output = document.getElementById('output'); const code = document.getElementById('code').value; const lang = document.getElementById('language').value; output.innerText = "Đang biên dịch..."; try { const response = await fetch('https://emkc.org/api/v2/piston/execute', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ language: lang, version: "*", files: [{ content: code }] }) }); const data = await response.json(); // --- ĐÂY LÀ PHẦN SỬA LỖI --- if (data.run) { // Nếu có lỗi biên dịch/thực thi if (data.run.stderr) { output.innerText = "Lỗi:\n" + data.run.stderr; } // Nếu chạy thành công và có output else if (data.run.stdout) { output.innerText = data.run.stdout; } // Trường hợp chạy thành công nhưng code không in ra gì else { output.innerText = "Chương trình chạy xong nhưng không có output."; } } else { output.innerText = "Không nhận được phản hồi từ máy chủ."; console.log(data); // In ra console trình duyệt để debug } } catch (error) { output.innerText = "Lỗi kết nối tới Piston API: " + error.message; } }
